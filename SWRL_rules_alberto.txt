dbo:Anime(?a) ^ 
schema:genre(?a, ?g) ^
dct:title (?g, ?genreName) ^ 
swrlb:containsIgnoreCase(?genreName, "Horror") -> suitableForKids(?a, false)

dbo:Anime(?a) ^ 
schema:genre(?a, ?g) ^
dct:title (?g, ?genreName) ^ 
swrlb:containsIgnoreCase(?genreName, "Love") -> suitableForKids(?a, false)

dbo:Anime(?a) ^ 
schema:genre(?a, ?g) ^
dct:title (?g, ?genreName) ^ 
swrlb:containsIgnoreCase(?genreName, "Suspense") -> suitableForKids(?a, false)
