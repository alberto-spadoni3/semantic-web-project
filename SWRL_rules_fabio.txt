# rule 3
# NB: solo la prima funziona, Protege non mi permette di usare il NOT (¬) in nessun modo

# se anime ha entrambe le date => finished airing
AnimeSeason(?anime) ^ airingStartDate(?a, ?start_date) ^ airingEndDate(?a, ?end_date) ->  creativeWorkStatus(?anime, CreativeWorkStatus_Finished_Airing)

# se anime ha solo data inizio => currently airing
AnimeSeason(?anime) ^ airingStartDate(?a, ?start_date) ^ ¬airingEndDate(?a, ?end_date) ->  creativeWorkStatus(?anime, CreativeWorkStatus_Currently_Airing)

# se anime non ha date => not yet aired
AnimeSeason(?anime) ^ ¬airingStartDate(?a, ?start_date) ^ ¬airingEndDate(?a, ?end_date) ->  creativeWorkStatus(?anime, CreativeWorkStatus_Not_Yet_Aired)
